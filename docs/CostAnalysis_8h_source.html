106  /// \brief Given a kind of analysis and a PModule it returns a instance of
107  /// this analysis.
108  std::unique_ptr<CostAnalysis>
109  createCostAnalysis(CostAnalysisKind Kind, const ProfileModule *PModule) {
110  switch (Kind) {
111  case RandomCost:
112  return std::unique_ptr<CostAnalysis>(new RandomAnalysis());
113  case InstructionCache:
114  return std::unique_ptr<CostAnalysis>(
115  new InstructionCacheAnalysis(PModule));
116  case RegisterUse:
117  return std::unique_ptr<CostAnalysis>(
118  new RegisterUseAnalysis(PModule));
119  default:
120  case StaticInstruction:
121  return std::unique_ptr<CostAnalysis>(new StaticInstructionAnalysis());
122  case TTIInstruction:
123  return std::unique_ptr<CostAnalysis>(new TTIInstructionAnalysis());
124  case Branch:
125  return std::unique_ptr<CostAnalysis>(new BranchAnalysis());
126  case Call:
127  return std::unique_ptr<CostAnalysis>(new CallAnalysis());
128  }
129  }
