33 /// \brief This class is a interface to implement a execution time measurer. The
34 /// way that this execution is done can be implemented in diferents ways.
35 class ExecutionTimeMeasurer {
36  private:

38  /// that was given by the constructor.
39  virtual void run(std::vector<std::string>&, char* const*) = 0;

41  /// \brief Clean the data cache in order to give a better result for the
42  /// execution time.
43  void flushCache() {
44  int cs = 32770 * 1024 / sizeof(double);
45  double* flush = (double*) calloc (cs, sizeof(double));
46  int i;
47  double tmp = 0.0;
48  for (i = 0; i < cs; i++)
49  tmp += flush[i];
50  assert (tmp <= 10.0);
51  free (flush);
52  }
53  public:
54  /// \brief Call and measure the execution time of the run function passing
55  /// the arguments argv and evnp to the module's main function.
56  double getExecutionTime(std::vector<std::string>& Argv,
57  char* const* Envp) {
58  flushCache();
59  timestampType T0 = getTimestamp();
60  run(Argv, Envp);
61  timestampType T1 = getTimestamp();
62  return (T1 - T0) / 1000000.0L;
63  }
