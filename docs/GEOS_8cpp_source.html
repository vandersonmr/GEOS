85 ProfileModule*
86 GEOS::applyPasses(const ProfileModule& PModule, PassSequence &PS) {
87  pid_t pid = fork();
88  if (pid == 0) {
89  Module *MyModule = PModule.getLLVMModule();
90  PassManager PM;
91  FunctionPassManager FPM(MyModule);
92  PS.populatePassManager(PM, FPM);
93 
94  for (auto &Func : *MyModule)
95  FPM.run(Func);
96 
97  PM.run(*MyModule);
98  PModule.print(".tmp");
99  exit(0);
100  }
101 
102  int ReturnStatus;
103  waitpid(pid, &ReturnStatus, 0);
104  if (ReturnStatus != 0) {
105  printf("Houston, We've Got a Problem! Our child have died!\n");
106  return PModule.getCopy();
107  } else {
108  LLVMContext &Context = getGlobalContext();
109  SMDiagnostic Error;
110  Module *MyModule =
111  parseIRFile(".tmp", Error, Context).release();
112  ProfileModule *ModuleCopy = new ProfileModule(MyModule);
113  ModuleCopy->repairProfiling();
114  ModuleCopy->setPasses(PS);
115  return ModuleCopy;
116  }
117 }
