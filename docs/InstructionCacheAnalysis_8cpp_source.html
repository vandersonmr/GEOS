158 AbstractCacheState* MachineAST::
159 calculateFuncInstanceCost(AbstractCacheState* EntryState, MachineFunction &MF) {
160  if (MF.size() == 0) return nullptr;
161  Visited[MF.getName().str()] = true;
162 
163  uint64_t Misses = 0;
164  int MaxDeep = 10;
165  bool modified = true;
166  while (modified) {
167  if ((--MaxDeep) == 0)
168  break;
169 
170  modified = false;
171  bool First = true;
172  for (auto &MB : MF) {
173  if (First) {
174  Outputs[&MB] = EntryState;
175  First = false;
176  }
177 
178  if (Outputs.count(&MB) == 0) {
179  modified = true;
180  Outputs[&MB] = new AbstractCacheState(S, W, L);
181  }
182 
183  MachineBasicBlock *MostFreqMB = nullptr;
184  unsigned MostFreqV = 0;
185  for (auto &Pred : MB.predecessors()) {
186  if (Outputs.count(Pred) != 0) {
187  uint64_t F = Profile->getBasicBlockFrequency(*(MB.getBasicBlock()));
188  if (F > MostFreqV) {
189  MostFreqV = F;
190  MostFreqMB = Pred;
191  }
192  }
193  }
194 
195  if (MostFreqMB != nullptr && Outputs.count(MostFreqMB) != 0) {
196  for (unsigned i = 0; i < S; i++) {
197  std::vector<unsigned> Tmp;
198  for (auto &Line : Outputs[MostFreqMB]->Sets[i].Cache)
199  Tmp.push_back(Line.first);
200 
201  for (auto &Line : Tmp) {
202  if (!Outputs[&MB]->isCached(Line))
203  modified = true;
204  Outputs[&MB]->Sets[i].cache(Line);
205  }
206  }
207  }
208 
209  auto I = MB.getBasicBlock()->begin();
210  for (auto &MI : MB) {
211  if (!Outputs[&MB]->isCached(Addresses[&MI])) {
212  modified = true;
213  Misses += Profile->getBasicBlockFrequency(*(MB.getBasicBlock()));
214  }
215 
216  if (MI.isCall()) {
217  while (!isa<CallInst>(I)) I++;
218  auto CalledFunc = cast<CallInst>(I)->getCalledFunction();
219  if (CalledFunc != nullptr) {
220  auto CalledMachineFunc = FuncToMachine[CalledFunc->getName().str()];
221  if (CalledMachineFunc != nullptr) {
222  auto CallName = CalledFunc->getName() + MF.getName();
223  if (CallTabuList.count(CallName.str()) == 0) {
224  CallTabuList[CallName.str()] = true;
225  calculateFuncInstanceCost(Outputs[&MB], *CalledMachineFunc);
226  }
227  }
228  }
229  }
230 
231  Outputs[&MB]->cache(Addresses[&MI]);
232  }
233 
234  }
235  }
236 
237  MFCost[MF.getName().str()] += Misses * 0.1;
238  return Outputs[&MF.back()];
239 }
