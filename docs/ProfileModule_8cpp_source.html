240 void ProfileModule::repairFunctionProfiling(Function *Func) {
241  std::unordered_map<BasicBlock*, bool> HasFreq;
242  for (auto &BB : *Func)
243  if (hasBasicBlockFrequency(BB))
244  HasFreq[&BB] = true;
245 
246  bool changed = true;
247  while(changed) {
248  changed = false;
249  for (auto &BB : *Func) {
250  if (HasFreq.count(&BB) == 0) {
251  uint64_t PredecessorsWeight = getExecutionFreqUsingPredecessors(&BB);
252  uint64_t SuccessorsWeight = getExecutionFreqUsingSuccessors(&BB);
253 
254  uint64_t ExecutionFreq = std::max(PredecessorsWeight, SuccessorsWeight);
255 
256  if (ExecutionFreq > getBasicBlockFrequency(BB) &&
257  ExecutionFreq != 0) {
258  setBasicBlockFrequency(BB, ExecutionFreq);
259  changed = true;
260  }
261  }
262  }
263  }
264 }

266 void ProfileModule::repairProfiling() {
267  for (auto &Func : *LLVMModule) {
268  for (auto &BB : Func) {
269  if (hasBranchFrequency(BB)) {
270  auto Frequency = 0;
271  auto NumSucc = 0;
272  for (auto &BFreq : getBranchFrequency(BB)) {
273  Frequency += BFreq;
274  auto Succ = BB.getTerminator()->getSuccessor(NumSucc);
275  if (!hasBasicBlockFrequency(*Succ))
276  setBasicBlockFrequency(*Succ, BFreq);
277  ++NumSucc;
278  }
279  setBasicBlockFrequency(BB, Frequency);
280  }
281  }
282  repairFunctionProfiling(&Func);
283  }
