62 void MachineRegisterUse::calculateRegisterCost(MachineFunction &MF) {
63  for (auto &MB : MF) {
64  int MBCost = 0;
65  for (auto &MI : MB)
66  if (MI.mayLoad() || MI.mayStore())
67  MBCost += 1;
68 
69  MFCost[MF.getName().str()] +=
70  Profile->getBasicBlockFrequency(*(MB.getBasicBlock())) * MBCost * 1.3;
71  }
72 }
