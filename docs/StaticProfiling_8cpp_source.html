79 std::list<Function*>& topologicalSort(CallGraph &CG) {
80  std::list<Function*> &SortedList = *(new std::list<Function*>());
81  std::unordered_map<Function*, Mark> Marks;
82 
83  for (auto &F : CG.getModule())
84  Marks[&F] = Mark::None;
85 
86  auto Fi = std::find_if(Marks.begin(), Marks.end(), isUnmarked);
87  while (Fi != Marks.end()) {
88  visit(*CG[Fi->first], Marks, SortedList);
89  Fi = std::find_if(Marks.begin(), Marks.end(), isUnmarked);
90  }
91

95 void spreadOverCallGraph(ProfileModule *Profile) {

97 
98  auto M = Profile->getLLVMModule();
99 
100  CallGraph CG(*M);
101  std::list<Function*> &Sorted = topologicalSort(CG);
102  CallWeights["main"] = 1;
103 
104  for (auto F : Sorted) {
105  for (auto &BB : *F) {
106  auto ExecutionFreq = Profile->getBasicBlockFrequency(BB);
107  for (auto &I : BB) {
108  if (isa<CallInst>(I)) {
109  Function *FuncCalled = cast<CallInst>(I).getCalledFunction();
110 
111  if (FuncCalled == nullptr)
112  continue;
113 
114  CallWeights[FuncCalled->getName().str()] +=
115  CallWeights[F->getName().str()] * ExecutionFreq;
116  }
117  }
118  }
119  }
120 
121  for (auto FuncFreq : CallWeights) {
122  Function &F = *(M->getFunction(FuncFreq.first));
123  for (auto &BB : F)
124  Profile->setBasicBlockFrequency(BB,
125  Profile->getBasicBlockFrequency(BB) * FuncFreq.second * 1000);
126  }
