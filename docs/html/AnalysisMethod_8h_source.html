68 /// \brief This method uses a weight given by the execution time of a similar
69 /// basic block. For that it uses a database of basic blocks and their execution
70 /// time. The search on the database in made using weights for each part of the
71 /// hash. For instance, double multiplication has a higher weight when looking
72 /// for similar basic blocks. Finally, this weight is multiplied by the
73 /// execution frequency to evaluate the execution cost of the given function.
74 class HashWeightedMethod : public AnalysisMethod {
75  private:
76  /// \brief It needs a basic block database.
77  DatabaseManager *DBManager;
78  public:
79  /// \brief The constructor needs a string with the path to the database

82  double estimateExecutionTime(llvm::Function*,
83  std::unique_ptr<llvm::GCOVFunction,
84  std::default_delete<llvm::GCOVFunction> >&);
85 };
