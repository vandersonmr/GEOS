63 DatabaseManager::getNearest(BBDescriptor &Hash, bool Weight = false) {
64  auto SmallestDistance = 1000000000000;
65  std::string Smallest = "";
66 
67  for (auto Elem : DB) {
68  BBDescriptor H(Elem.first);
69  auto Distance = 0;
70  if (Weight)
71  Distance = BBDescriptor::distance(H, Hash, true);
72  else
73  Distance = BBDescriptor::distance(H, Hash, false);
74 
75  if (Distance == 0) { Smallest = Elem.first; break; }
76  if (Distance < SmallestDistance) {
77  SmallestDistance = Distance;
78  Smallest = Elem.first;
79  }
80  }
81
