83 BBHash* DatabaseManager::
84 getNearest(const BBHash &Hash) const {
85  if (Nearest.count(Hash) == 0) {
86  double SmallestDistance = std::numeric_limits<double>::max();
87  llvm::StringRef Smallest ="";
88 
89  int dotProduct = BBHash::dotProduct(Hash, *RandomHash);
90  auto Hashes = CDB.at((dotProduct/w) % PrimeNumber);
91 
92  for (auto &Elem : Hashes) {
93  double Distance = 0;
94  Distance = BBHash::distance(Elem, Hash);
95 
96  if (Distance == 0) {
97  Smallest = Elem.getString();
98  break;
99  } if (Distance < SmallestDistance) {
100  SmallestDistance = Distance;
101  Smallest = Elem.getString();
102  }
103  }
104 
105  Nearest[Hash] = new BBHash(Smallest);
106  }
107 
108  return Nearest[Hash];
109 }
