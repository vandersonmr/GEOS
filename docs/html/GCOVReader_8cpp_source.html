25 std::vector<GCOVFunction*>
26 readFunctions(GCOVFile& GF, GCOVBuffer &GCNOBuffer, GCOVBuffer &GCDABuffer) {
27  GCOV::GCOVVersion Version;
28  std::vector<GCOVFunction*> *Functions =
29  new std::vector<GCOVFunction*>;
30 
31  uint32_t Checksum;
32 
33  GCOVOptions Options(true, true, true, true, true, true, false, false);
34  FileInfo FI(Options);
35 
36  if (!GCNOBuffer.readGCNOFormat() || !GCDABuffer.readGCDAFormat())
37  return *Functions;
38  if (!GCNOBuffer.readGCOVVersion(Version) ||
39  !GCDABuffer.readGCOVVersion(Version))
40  return *Functions;
41 
42  if (!GCNOBuffer.readInt(Checksum) || !GCDABuffer.readInt(Checksum))
43  return *Functions;
44 
45  while (true) {
46  if (!GCNOBuffer.readFunctionTag()) break;
47  if (!GCDABuffer.readFunctionTag()) break;
48  auto GFun = new GCOVFunction(GF);
49  if (!GFun->readGCNO(GCNOBuffer, Version) ||
50  !GFun->readGCDA(GCDABuffer, Version))
51  return *Functions;
52 
53  GFun->collectLineCounts(FI);
54 
55  Functions->push_back(GFun);
56  }
57 
58  FI.setRunCount(1);
59  FI.setProgramCount(1);
60 
61  return *Functions;
62 }

64 void loadGCOV(std::vector<MemoryBuffer*> GCDAs,
65  std::vector<MemoryBuffer*> GCNOs, ProfileModule *Profile) {
66 
67  auto I = 0;
68  for (auto &Func : *Profile->getLLVMModule())

71

73  std::vector<MemoryBuffer*>::iterator iGCNO = GCNOs.begin();
74 
75  while (iGCDA != GCDAs.end() && iGCNO != GCNOs.end()) {
76  llvm::GCOVFile GF;
77 
78  GCOVBuffer GCNO_GB(*iGCNO);
79  GCOVBuffer GCDA_GB(*iGCDA);
80 
81  std::vector<GCOVFunction*> FuncsProfile =
82  readFunctions(GF, GCNO_GB, GCDA_GB);
83 
84  auto LLVMModule = Profile->getLLVMModule();
85 
86  for (auto &FreqFunc : FuncsProfile) {
87  Function *LLVMFunc = LLVMModule->getFunction(FreqFunc->getName());
88  assert(LLVMFunc != nullptr
89  && "Trying to access a LLVM Function that don't exist!");
90 
91  auto MBB = FreqFunc->block_begin();
92  ++MBB;
93  ++MBB;
94  for(auto &BB : *LLVMFunc) {
95  Profile->setBasicBlockFrequency(BB, (*MBB).get()->getCount());
96  (*MBB)->sortDstEdges();
97  std::vector<uint32_t> Freqs;
98  for (auto BranchFreq = (*MBB)->dst_begin();
99  BranchFreq != (*MBB)->dst_end(); ++BranchFreq) {
100  Freqs.push_back((*BranchFreq)->Count);
101  }
102 
103  if (Freqs.size() >= 2)
104  Profile->setBranchFrequency(BB, Freqs);
105 
106  ++MBB;
107  }
108  }
109 
110  ++iGCDA;
111  ++iGCNO;
112  }
