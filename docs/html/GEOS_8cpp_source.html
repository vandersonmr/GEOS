25 AnalysisMethod*
26 GEOS::getAnalyser(AnalysisMethodKind Method, StringRef DatabaseFilename = "") {
27  AnalysisMethod *Analyser;
28 
29  switch (Method) {
30  case hashWM:
31  if (!DatabaseFilename.empty())
32  Analyser = new HashWeightedMethod(DatabaseFilename);
33  else
34  errs()
35  << "For hashM you must pass a database filename with -database\n";
36  break;
37  case instM:
38  Analyser = new InstructionMethod();
39  break;
40  case instCostM:
41  Analyser = new InstructionCostMethod();
42  break;
43  case randM:
44  Analyser = new RandomMethod();
45  break;
46  case freqM:
47  Analyser = new FrequencyMethod();
48  break;
49  case hashPlusInstM:
50  if (!DatabaseFilename.empty())
51  Analyser = new InstructionPlusHashMethod(DatabaseFilename);
52  else
53  errs() << "For hashPlusInstM you must pass a database filename with" <<
54  " -database\n";
55  break;
56  default:
57  Analyser = new InstructionMethod();
58  break;
59  }
60 
61  return Analyser;
62 }
