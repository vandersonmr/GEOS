239  if (RandomSize)

280  /// \brief Returns a PassSequence of the same size of both PassSequences
281  /// (parameter and this). This PassSequences is created by a mix between
282  /// them. Where there is 50% chance for each pass to be from the first or
283  /// second sequence.
284  PassSequence crossOver(const PassSequence &Rhs) const {
285  PassSequence Res;
286  auto Irhs = Rhs.begin();
287  for (auto Opt : Opts) {
288  int i = getRandom(1,10);
289  if (i % 2 == 0) {
290  Res.add(Opt);

292  Res.add(*Irhs);
293  }
294  Irhs++;
295  }

297  }
