100 ProfileModule::getExecutionFreqUsingPredecessors(BasicBlock* BB) {
101  uint64_t PredecessorsWeight = 0;
102  for (auto IT = pred_begin(BB), ET = pred_end(BB); IT != ET; ++IT) {
103  BasicBlock* Predecessor = *IT;
104  auto Terminator = Predecessor->getTerminator();
105 
106  if (BB != Predecessor) {
107  if (isa<BranchInst>(Terminator)) {
108  if (cast<BranchInst>(Terminator)->isUnconditional()) {
109 
110  PredecessorsWeight += getBasicBlockFrequency(Predecessor);
111 
112  } else {
113 
114  uint64_t EdgeWeight = getBasicBlockFrequency(Predecessor);
115  for (unsigned J = 0; J < Terminator->getNumSuccessors(); ++J) {
116  auto SuccPred = Terminator->getSuccessor(J);
117  if (SuccPred != BB) {
118  if (!hasBasicBlockFrequency(SuccPred) ||
119  SuccPred->getUniquePredecessor() == nullptr ||
120  EdgeWeight < getBasicBlockFrequency(SuccPred)) {
121  EdgeWeight = 0;
122  break;
123  } else {
124  EdgeWeight -= getBasicBlockFrequency(SuccPred);
125  }
126  }
127  }
128 
129  PredecessorsWeight += EdgeWeight;
130  }
131  }
132  }
133 
134  }
135 
136  return PredecessorsWeight;
137 }

183 void ProfileModule::repairFunctionProfiling(Function *Func) {
184  bool changed = true;
185  while(changed) {
186  changed = false;
187  for (auto &BB : *Func) {
188  if (!BB.hasName())
189  BB.setName(std::to_string(++I));
190 
191  uint64_t PredecessorsWeight = getExecutionFreqUsingPredecessors(&BB);
192  uint64_t SuccessorsWeight = getExecutionFreqUsingSuccessors(&BB);
193 
194  uint64_t ExecutionFreq = std::max(PredecessorsWeight, SuccessorsWeight);
195 
196  if (ExecutionFreq > getBasicBlockFrequency(&BB) &&
197  ExecutionFreq != 0) {
198  setBasicBlockFrequency(&BB, ExecutionFreq);
199  changed = true;
200  }
201  }
202  }
203 }
