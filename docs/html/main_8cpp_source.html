63 readFunctions(GCOVFile& GF, GCOVBuffer &GCNOBuffer, GCOVBuffer &GCDABuffer) {
64  GCOV::GCOVVersion Version;
65  std::vector<std::unique_ptr<GCOVFunction>> Functions;
66  uint32_t Checksum;
67 
68  GCOVOptions Options(true, true, true, true, true, true, false, false);
69  FileInfo FI(Options);
70 
71  if (!GCNOBuffer.readGCNOFormat() || !GCDABuffer.readGCDAFormat())
72  return Functions;
73  if (!GCNOBuffer.readGCOVVersion(Version) ||
74  !GCDABuffer.readGCOVVersion(Version))
75  return Functions;
76 
77  if (!GCNOBuffer.readInt(Checksum) || !GCDABuffer.readInt(Checksum))
78  return Functions;
79 
80  while (true) {
81  if (!GCNOBuffer.readFunctionTag()) break;
82  if (!GCDABuffer.readFunctionTag()) break;
83  auto GFun = make_unique<GCOVFunction>(GF);
84  if (!GFun->readGCNO(GCNOBuffer, Version) ||
85  !GFun->readGCDA(GCDABuffer, Version))
86  return Functions;
87 
88  GFun->collectLineCounts(FI);
89 
90  Functions.push_back(move(GFun));
91  }
92 
93  FI.setRunCount(1);
94  FI.setProgramCount(1);
95 
96  return Functions;
97 }
