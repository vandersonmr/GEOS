30 /* Definition of PAPI_VERSION format. Note that the minor version and
31  * revision numbers _must_ be less than 256. Also, the PAPI_VER_CURRENT
32  * masks out the revision. Any revision change is supposed to be binary
33  * compatible between the user application code and the run-time library.
34  * Any modification that breaks this compatibility _should_ modify the
35  * minor version number as to force user applications to re-compile
36  */
37 #define PAPI_VERSION_NUMBER(maj,min,rev) (((maj)<<16) | ((min)<<8) | (rev))
38 #define PAPI_VERSION_MAJOR(x) (((x)>>16) & 0xffff)
39 #define PAPI_VERSION_MINOR(x) (((x)>>8) & 0xff)
40 #define PAPI_VERSION_REVISION(x) ((x) & 0xff)

487 /*
488  This structure is the event information that is exposed to the user through the API.
489  The same structure is used to describe both preset and native events.
490  WARNING: This structure is very large. With current definitions, it is about 2660 bytes.
491  Unlike previous versions of PAPI, which allocated an array of these structures within
492  the library, this structure is carved from user space. It does not exist inside the library,
493  and only one copy need ever exist.
494  The basic philosophy is this:
495  - each preset consists of a code, some descriptors, and an array of native events;
496  - each native event consists of a code, and an array of register values;
497  - fields are shared between preset and native events, and unused where not applicable;
498  - To completely describe a preset event, the code must present all available
499  information for that preset, and then walk the list of native events, retrieving
500  and presenting information for each native event in turn.
501  The various fields and their usage is discussed below.
502 */
503 /* MAX_TERMS is the current max value of MAX_COUNTER_TERMS as defined in SUBSTRATEs */
504 #define PAPI_MAX_INFO_TERMS 8
505  typedef struct event_info {
506  unsigned int event_code; /* preset (0x8xxxxxxx) or native (0x4xxxxxxx) event code */
507  unsigned int count; /* number of terms (usually 1) in the code and name fields
508  - for presets, these terms are native events
509  - for native events, these terms are register contents */
510  char symbol[PAPI_MAX_STR_LEN+3]; /* name of the event
511  - for presets, something like PAPI_TOT_INS
512  - for native events, something related to the vendor name */
513  char short_descr[PAPI_MIN_STR_LEN]; /* a description suitable for use as a label, typically only
514  implemented for preset events */
515  char long_descr[PAPI_HUGE_STR_LEN]; /* a longer description of the event
516  - typically a sentence for presets
517  - possibly a paragraph from vendor docs for native events */
518  char derived[PAPI_MIN_STR_LEN]; /* name of the derived type
519  - for presets, usually NOT_DERIVED
520  - for native events, empty string
521  NOTE: a derived description string is available
522  in papi_data.c that is currently not exposed to the user */
523  char postfix[PAPI_MIN_STR_LEN]; /* string containing postfix operations; only defined for
524  preset events of derived type DERIVED_POSTFIX */
525  unsigned int code[PAPI_MAX_INFO_TERMS];/* array of values that further describe the event:
526  - for presets, native event_code values
527  - for native events, register values for event programming */
